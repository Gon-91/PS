#include<iostream>
using namespace std;
 
 
// 전역변수 - 임의의 배열 생성
int Data[1000] ;
int N;
 
 
// 버블 정렬 함수
/*
    n개의 배열에서 (예제의 경우 10개)
    가장 첫번째 배열(기준점)부터 n-1 개의 배열을 각각 비교하여
    기준점이 되는 배열의 수가 다음 배열보다 크다면 값을 서로 바꿔준다.
    이 과정을 n-1만큼 반복하면 정렬이 완성된다.
*/
void Bouble_Sort()
{
 
    for (int i = 0; i < N; i++)
    {
        for (int j = 0; j < N-1; j++)
        {
            if (Data[j] > Data[j+1])
            {
                int temp = 0;
                temp = Data[j];
                Data[j] = Data[j+1];
                Data[j+1] = temp;
            }
        }
    }
}
 
// 프로그램의 구동 및 출력
int main()
{
    // Test 배열의 갯수 입력 및 배열의 Data 입력
    cin >> N;
    for (int i = 0; i < N; i++)
    {
        cin >> Data[i];
    }
 
    //Bouble sort 시작
    Bouble_Sort();
 
    //Bouble sort 가 완료된 이후 정렬된 배열의 출력.
    for (int i = 0; i < N; i++)
    {
        cout << Data[i] << endl;
    }
    return 0;
}
